# 坐标数据格式
## 点坐标格式
````json
[120, 30]
````
## 线坐标格式
````json
[
    [120.5, 30.5],
    [120.5, 29]
]
````
## 面坐标格式
````json
[
    [
        [121, 31],
        [121.5, 31],
        [121.5, 32],
        [121, 32]
    ]
]
````
带内环洞的面坐标
````json
[
    [
        [121, 31],
        [121.5, 31],
        [121.5, 32],
        [121, 32]
    ],
    // 内环洞
    [
        [121.2, 31.4],
        [121.3, 31.4],
        [121.25, 31.7]
    ]
]
````
## 多点坐标格式
````json
[
    // 第一个点
    [120.5, 30.5],
    // 第二个点
    [120.5, 29]
]
````
## 多线坐标格式
````json
[
    // 第一条线
    [
        [120.5, 30.5],
        [120.5, 29]
    ],
    // 第二条线
    [
        [121, 31],
        [121, 32],
        [122, 32]
    ]
]
````
## 多面坐标格式
````json
[
    // 第一个面
    [
        [
            [121, 31],
            [121.5, 31],
            [121.5, 32],
            [121, 32]
        ],
        // 内环洞
        [
            [121.2, 31.4],
            [121.3, 31.4],
            [121.25, 31.7]
        ]
    ],
    // 第二个面
    [
        [
            [122, 31],
            [122.5, 31],
            [122.5, 32],
            [122, 32]
        ]
    ]
]
````
